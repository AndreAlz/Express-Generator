import { Roles } from "../../entity/Roles";

export class TokenDataDto {
  private _id_persona: string;
  private _id_usuario: string;
  public _roles: Array<Roles>;

  constructor(id_persona: string, id_usuario: string, roles: Array<Roles>) {
    this._id_persona = id_persona;
    this._id_usuario = id_usuario;
    this._roles = roles;
  }

  /**
   * Getter id_persona
   * @return {string}
   */
  public get id_persona(): string {
    return this._id_persona;
  }

  /**
   * Getter id_usuario
   * @return {string}
   */
  public get id_usuario(): string {
    return this._id_usuario;
  }

  /**
   * Getter roles
   * @return {Array<Roles>}
   */
  public get roles(): Array<Roles> {
    return this._roles;
  }

  /**
   * Setter id_persona
   * @param {string} value
   */
  public set id_persona(value: string) {
    this._id_persona = value;
  }

  /**
   * Setter id_usuario
   * @param {string} value
   */
  public set id_usuario(value: string) {
    this._id_usuario = value;
  }

  /**
   * Setter roles
   * @param {Array<Roles>} value
   */
  public set roles(value: Array<Roles>) {
    this._roles = value;
  }
}
